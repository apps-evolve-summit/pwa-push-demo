app.post('/api/send-push', (req, res) => {
    var allSubscriptions = getAllSubscriptions();

    allSubscriptions.forEach(subscriber => {
        webpush.sendNotification(
            subscriber,
            req.body.data,
            options).catch((err) => {
                console.log('Error while pushing...');
            });
    });
    res.status(200).send({success: true});
});
